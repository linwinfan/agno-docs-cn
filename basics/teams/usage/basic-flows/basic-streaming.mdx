<Steps>

<Step title="创建 Python 文件">
    ```bash
    touch basic_streaming.py
    ```
  </Step>

  <Step title="将以下代码添加到您的 Python 文件中">

    ```python basic_streaming.py
    from typing import Iterator  # noqa
    from agno.agent import Agent
    from agno.models.openai import OpenAIChat
    from agno.team import Team
    from agno.tools.exa import ExaTools

    # 股价和分析师数据智能体
    stock_searcher = Agent(
        name="股票搜索器",
        model=OpenAIChat(id="gpt-5-mini"),
        role="在网上搜索股票信息。",
        tools=[
            ExaTools(
                include_domains=["cnbc.com", "reuters.com", "bloomberg.com", "wsj.com"],
                text=False,
                show_results=True,
                highlights=False,
            )
        ],
    )

    # 公司信息智能体
    company_info_agent = Agent(
        name="公司信息搜索器",
        model=OpenAIChat(id="gpt-5-mini"),
        role="在网上搜索股票信息。",
        tools=[
            ExaTools(
                include_domains=["cnbc.com", "reuters.com", "bloomberg.com", "wsj.com"],
                text=False,
                show_results=True,
                highlights=False,
            )
        ],
    )

    # 创建具有流式传输能力的团队
    team = Team(
        name="股票研究团队",
        model=OpenAIChat(id="gpt-5-mini"),
        members=[stock_searcher, company_info_agent],
        markdown=True,
        show_members_responses=True,
    )

    # 测试流式响应
    team.print_response(
        "What is the current stock price of NVDA?",
        stream=True,
    )
    ```

  </Step>
